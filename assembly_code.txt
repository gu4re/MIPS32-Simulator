# New little changes not in previous versions
# [17/10/2023] Modified msg1 to inform N > 1 to avoid infinite loop
# [18/10/2023] Modified $t0 initial value from 25 to 5 so the program is not too long
# [18/10/2023] Modified beq instruction to bge avoiding infinite loop
# Various labels
        .data
msg1:	.asciiz	"Number of integers (N > 1)? "
msg2:   .asciiz "Number found: "
Z:      .word 0

# The label 'main' represents the starting point
        .text
main:
        # Print msg1
	    li	    $v0, 4		# print_string syscall code = 4
	    la	    $a0, msg1
	    syscall
	    # Get N from user and save
	    li	    $v0,5		# read_int syscall code = 5
	    syscall
	    addi	$t1, $v0, 0	# syscall results returned in $v0
	    li      $t0,5
	    mul     $t0, $t1,$t0
	    sw      $t0,Z
	    sub     $t0,$t0,$t1
	    li	    $t2,0		# initialize counter (i)
	    li	    $t3,0		# initialize sum
	    lw      $t4,Z
# Main loop body
loop:
        addi    $t2,$t2,1 # i = i + 1
        add     $t3,$t3,$t2 # sum = sum + i
        bge     $t0,$t3,exit # if value = sum of iterations
        j       loop
# Exit routine found - print msg2
exit:
        li	    $v0,4		# print_string syscall code = 4
        la	    $a0,msg2
        syscall
        # Print result
        li	    $v0,1		# print_string syscall code = 4
        addi	$a0,$t3,0
        syscall
